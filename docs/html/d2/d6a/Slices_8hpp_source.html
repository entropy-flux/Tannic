38template <Expression Source, class... Indexes>
39class Slice {
40public:
41 constexpr Slice(typename Trait<Source>::Reference source, std::tuple<Indexes...> indexes)
42 : dtype_(source.dtype())
43 , source_(source)
44 , indexes_(indexes)
45 {
46 std::array<Shape::size_type, Shape::limit> shape{};
47 std::array<Strides::size_type, Strides::limit> strides{};
48 Shape::rank_type dimension = 0;
49 Shape::rank_type rank = 0;
50 offset_ = 0;
51 auto process = [&](const auto& argument) {
52 using Argument = std::decay_t<decltype(argument)>;
53 if constexpr (std::is_same_v<Argument, indexing::Range>) {
54 auto range = normalize(argument, source.shape()[dimension]);
55 auto size = range.stop - range.start;
56 assert(size >= 0);
57 shape[rank] = size;
58 strides[rank] = source.strides()[dimension];
59 offset_ += range.start * source.strides()[dimension] * dsizeof(dtype_);
60 rank++; dimension++;
61 }
62
63 else if constexpr (std::is_integral_v<Argument>) {
64 auto index = indexing::normalize(argument, source.shape()[dimension]);
65 offset_ += index * source.strides()[dimension] * dsizeof(dtype_);
66 dimension++;
67 }
68
69 else {
70 static_assert(sizeof(Argument) == 0, "Unsupported index type in Slice");
71 }
72 };
73
74 std::apply([&](const auto&... arguments) {
75 (process(arguments), ...);
76 }, indexes);
77
78 while (dimension < source.rank()) {
79 shape[rank] = source.shape()[dimension];
80 strides[rank] = source.strides()[dimension];
81 rank++; dimension++;
82 }
83
84 shape_ = Shape(shape.begin(), shape.begin() + rank);
85 strides_ = Strides(strides.begin(), strides.begin() + rank);
86 }

155template <Expression Source, class... Indexes>
156template <typename T>
157void Slice<Source, Indexes...>::operator=(T value) {
158 auto copy = [this](std::byte const* value, std::ptrdiff_t offset) {
159 if(rank() == 0) {
160 assign(value, offset);
161 return;
162 }
163
164
165 std::vector<std::size_t> indexes(rank(), 0);
166 bool done = false;
167
168 while (!done) {
169 std::size_t position = offset;
170 for (auto dimension = 0; dimension < rank(); ++dimension) {
171 position += indexes[dimension] * strides_[dimension] * dsizeof(dtype_);
172 }
173
174 assign(value, position);
175 done = true;
176 for (int dimension = rank() - 1; dimension >= 0; --dimension) {
177 if (++indexes[dimension] < shape_[dimension]) {
178 done = false;
179 break;
180 }
181 indexes[dimension] = 0;
182 }
183 }
184 };
185
186 switch (dtype_) {
187 case int8: {
188 int8_t casted = value;
189 copy(tobytes(casted), offset());
190 break;
191 }
192 case int16: {
193 int16_t casted = value;
194 copy(tobytes(casted), offset());
195 break;
196 }
197 case int32: {
198 int32_t casted = value;
199 copy(tobytes(casted), offset());
200 break;
201 }
202 case int64: {
203 int64_t casted = value;
204 copy(tobytes(casted), offset());
205 break;
206 }
207 case float32: {
208 float casted = value;
209 copy(tobytes(casted), offset());
210 break;
211 }
212 case float64: {
213 double casted = value;
214 copy(tobytes(casted), offset());
215 break;
216 }
217 default:
218 break;
219 }
220}
