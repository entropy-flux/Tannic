31template<Integral Index, Integral Size>
32constexpr inline Index normalize(Index index, Size bound) {
33 if (index < 0) index += bound;
34 assert(index >= 0 && index < bound && "Index out of bounds");
35 return index;
36}

38template<Integral Size>
39constexpr inline Range normalize(Range range, Size size) {
40 int start = range.start < 0 ? size + range.start : range.start;
41 int stop = range.stop < 0 ? size + range.stop + 1 : range.stop;
42 return {start, stop};
43}
