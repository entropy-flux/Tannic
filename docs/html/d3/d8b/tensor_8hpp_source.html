56 Tensor(type dtype, Shape shape, Strides strides, std::ptrdiff_t offset = 0)
57 : dtype_(dtype)
58 , shape_(shape)
59 , strides_(strides)
60 , offset_(offset)
61 {
62 std::size_t expected = 1;
63 for (std::size_t dimension = 0; dimension < rank(); ++dimension) {
64 std::size_t index = rank() - 1 - dimension;
65 if (strides_[index] != expected) {
66 is_contiguous_ = false;
67 }
68 nelements_ *= shape_[index];
69 expected *= shape_[index];
70 }
71 }

327 template<Arithmetic T>
328 void initialize(std::initializer_list<T> values, Environment environment = Host{}) {
329 if (!is_contiguous())
330 throw Exception("Assign to initializer list supported only for contiguous tensors");
331
332 if (rank() != 1 || shape_[0] != values.size())
333 throw Exception("Shape mismatch in assignment from initializer_list");
334
335 if (!is_initialized())
336 initialize(environment);
337
338 if (dtype_ == boolean) {
339 std::ptrdiff_t index = 0;
340 for (auto const& value : values) {
341 assign((bool const*)(&value), index);
342 ++index;
343 }
344 }
345
346 else {
347 auto fill = [this, &values](auto cast) {
348 using Cast = decltype(cast);
349 size_t index = 0;
350 for (auto value : values) {
351 Cast casted(value);
352 assign(expression::tobytes(casted), index * dsizeof(dtype_));
353 ++index;
354 }
355 };
356
357 switch (dtype_) {
358 case int8: fill(int8_t{}); break;
359 case int16: fill(int16_t{}); break;
360 case int32: fill(int32_t{}); break;
361 case int64: fill(int64_t{}); break;
362 case float16: fill(float16_t{}) ; break;
363 case bfloat16: fill(bfloat16_t{}); break;
364 case float32: fill(float{}); break;
365 case float64: fill(double{}); break;
366 default: throw Exception("Unsupported dtype in assignment");
367 }
368 }
369 }
