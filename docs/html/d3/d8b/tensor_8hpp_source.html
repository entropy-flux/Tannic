128 Tensor(type dtype, Shape shape, Strides strides, std::ptrdiff_t offset = 0)
129 : dtype_(dtype)
130 , shape_(shape)
131 , strides_(strides)
132 , offset_(offset)
133 {
134 std::size_t expected = 1;
135 for (std::size_t dimension = 0; dimension < rank(); ++dimension) {
136 std::size_t index = rank() - 1 - dimension;
137 if (strides_[index] != expected) {
138 is_contiguous_ = false;

561 template<typename T>
562 void initialize(std::initializer_list<T> values, Environment environment = Host{}) {
563 if (!is_contiguous())
564 throw Exception("Assign to initializer list supported only for contiguous tensors");
565
566 if (rank() != 1 || shape_[0] != values.size())
567 throw Exception("Shape mismatch in assignment from initializer_list");
568
569 if (!is_initialized())
570 initialize(environment);
571
572 if (dtype_ == boolean) {
573 std::ptrdiff_t index = 0;
574 for (auto const& value : values) {
575 assign((bool const*)(&value), index);
576 ++index;
577 }
578 }
579
580 else {
581 auto fill = [this, &values](auto cast) {
582 using Cast = decltype(cast);
583 size_t index = 0;
584 for (auto value : values) {
585 Cast casted(value);
586 assign(expression::tobytes(casted), index * dsizeof(dtype_));
587 ++index;
588 }
589 };
590
591 switch (dtype_) {
592 case int8: fill(int8_t{}); break;
593 case int16: fill(int16_t{}); break;
594 case int32: fill(int32_t{}); break;
595 case int64: fill(int64_t{}); break;
596 case float16: fill(float16_t{}) ; break;
597 case bfloat16: fill(bfloat16_t{}); break;
598 case float32: fill(float{}); break;
599 case float64: fill(double{}); break;
600 default: throw Exception("Unsupported dtype in assignment");
601 }
602 }
603 }
