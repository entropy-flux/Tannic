108template<Integral Size>
109constexpr inline Range normalize(Range range, Size size) {
110 int start = range.start < 0 ? size + range.start : range.start;
111 int stop = range.stop < 0 ? size + range.stop + 1 : range.stop;
112
113 if (start < 0 | start > size | stop < 0 | stop > size)
114 throw Exception("Range out of bounds");
115 return {start, stop};
116}
