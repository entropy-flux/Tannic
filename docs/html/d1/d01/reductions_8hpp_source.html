63template<class Reducer, Composable Operand>
64class Reduction : public Expression<Reducer, Operand> {
65public:
66 constexpr Reduction(Reducer reducer, typename Trait<Operand>::Reference operand)
67 : Expression<Reducer, Operand>(reducer, operand)
68 , dtype_(reducer.reduce(operand.dtype()))
69 , shape_(reducer.reduce(operand.shape()))
70 , strides_(shape_) {}
71
72 constexpr type dtype() const {
73 return dtype_;
74 }
75
76 constexpr Shape const& shape() const {
77 return shape_;
78 }
79
80 constexpr Strides const& strides() const {
81 return strides_;
82 }
83
84 std::ptrdiff_t offset() const {
85 return 0;
86 }
87
88 Tensor forward(Context const& context) const {
89 Tensor source = std::get<0>(this->operands).forward(context);
90 Tensor result(this->dtype(),shape(), strides(), offset());
91 this->operation.forward(source, result);
92 return result;
93 }
94
95private:
96 type dtype_;
97 Shape shape_;
98 Strides strides_;
99};

101struct Argmax {
102 int axis;
103 bool keepdim;
104
105 constexpr type reduce(type dtype) const {
106 return int64;
107 }
108
109 constexpr Shape reduce(Shape const& shape) const {
110 if (shape.rank() == 0)
111 throw Exception("Cannot reduce scalar tensors");
112
113 Shape out;
114 for (size_t dim = 0; dim < shape.rank(); ++dim) {
115 if (dim != static_cast<size_t>(axis)) out.expand(shape[dim]);
116 else if (keepdim) out.expand(1);
117 }
118 return out;
119 }
120
121 void forward(Tensor const& input, Tensor& output) const;
122};
