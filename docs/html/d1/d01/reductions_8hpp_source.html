72template<class Reducer, Composable Operand>
73class Reduction : public Expression<Reducer, Operand> {
74public:
75 constexpr Reduction(Reducer reducer, typename Trait<Operand>::Reference operand)
76 : Expression<Reducer, Operand>(reducer, operand)
77 , dtype_(reducer.reduce(operand.dtype()))
78 , shape_(reducer.reduce(operand.shape()))
79 , strides_(shape_) {}
80
81 constexpr type dtype() const {
82 return dtype_;
83 }
84
85 constexpr Shape const& shape() const {
86 return shape_;
87 }
88
89 constexpr Strides const& strides() const {
90 return strides_;
91 }
92
93 std::ptrdiff_t offset() const {
94 return 0;
95 }
96
97 Tensor forward() const {
98 Tensor source = std::get<0>(this->operands).forward();
99 Tensor result(dtype(),shape(), strides(), offset());
100 this->operation.forward(source, result);
101 return result;
102 }
103
104private:
105 type dtype_;
106 Shape shape_;
107 Strides strides_;
108};

130struct Argmax {
131 int axis;
132 bool keepdim;
133
134 constexpr type reduce(type dtype) const {
135 return int64;
136 }
137
138 constexpr Shape reduce(Shape const& shape) const {
139 if (shape.rank() == 0)
140 throw Exception("Cannot reduce scalar tensors");
141
142 Shape out;
143 for (size_t dim = 0; dim < shape.rank(); ++dim) {
144 if (dim != static_cast<size_t>(axis)) out.expand(shape[dim]);
145 else if (keepdim) out.expand(1);
146 }
147 return out;
148 }
149
150 void forward(Tensor const& input, Tensor& output) const;
151};
