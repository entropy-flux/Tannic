39static constexpr Shape broadcast(Shape const& first, Shape const& second) {
40 auto first_rank = first.rank();
41 auto second_rank = second.rank();
42 auto rank = std::max(first_rank, second_rank);
43 std::vector<std::size_t> result(rank, 1);
44
45 for (auto dimension = 0; dimension < rank; ++dimension) {
46 auto first_dimension = (dimension < rank - first_rank) ? 1 : first[dimension - (rank - first_rank)];
47 auto second_dimension = (dimension < rank - second_rank) ? 1 : second[dimension - (rank - second_rank)];
48
49 if (!(first_dimension == second_dimension || first_dimension == 1 || second_dimension == 1)) {
50 throw std::invalid_argument("Shapes are not broadcast-compatible.");
51 }
52 result[dimension] = std::max(first_dimension, second_dimension);
53 }
54 return Shape(result);
55}
