280static constexpr Shape broadcast(Shape const& first, Shape const& second) {
281 auto first_rank = first.rank();
282 auto second_rank = second.rank();
283 auto rank = std::max(first_rank, second_rank);
284 std::vector<std::size_t> result(rank, 1);
285
286 for (auto dimension = 0; dimension < rank; ++dimension) {
287 auto first_dimension = (dimension < rank - first_rank) ? 1 : first[dimension - (rank - first_rank)];
288 auto second_dimension = (dimension < rank - second_rank) ? 1 : second[dimension - (rank - second_rank)];
289
290 if (!(first_dimension == second_dimension || first_dimension == 1 || second_dimension == 1)) {
291 throw std::invalid_argument("Shapes are not broadcast-compatible.");
292 }
293 result[dimension] = std::max(first_dimension, second_dimension);
294 }
295 return Shape(result);
296}
