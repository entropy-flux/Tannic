94 template<Integral... Indexes>
95 constexpr Reshape(Trait<Source>::Reference source, Indexes... indexes)
96 : shape_(indexes...)
97 , strides_(shape_)
98 , source_(source) {
99 std::size_t elements = 0;
100 for (std::size_t dimension = 0; dimension < sizeof...(indexes); ++dimension) {
101 elements += strides_[dimension] * (shape_[dimension] - 1);
102 }
103 elements += 1;
104 assert(elements == std::accumulate(shape_.begin(), shape_.end(), 1ULL, std::multiplies<>{})
105 && "Shape mismatch: view must preserve total number of elements");
106 }

115 constexpr type dtype() const {
116 return source_.dtype();
117 }

127 }

204 constexpr Transpose(typename Trait<Source>::Reference source, std::pair<int, int> dimensions)
205 : shape_(source.shape())
206 , strides_(source.strides())
207 , source_(source)
208 , dimensions_(dimensions) {
209 auto rank = source.shape().rank();
210 std::swap(shape_[indexing::normalize(dimensions.first, rank)], shape_[indexing::normalize(dimensions.second, rank)]);
211 std::swap(strides_[indexing::normalize(dimensions.first, rank)], strides_[indexing::normalize(dimensions.second, rank)]);
212 }
