53constexpr Shape transpose(Shape const& layout, std::pair<int, int> const& dimensions) {
54 auto rank = layout.rank();
55 std::array<Shape::size_type, Shape::limit> sizes{};
56 for (Shape::rank_type dimension = 0; dimension < rank; ++dimension) {
57 sizes[dimension] = layout[dimension];
58 }
59
60 std::swap(sizes[indexing::normalize(dimensions.first, rank)], sizes[indexing::normalize(dimensions.second, rank)]);
61 return Shape(sizes.begin(), sizes.begin() + rank);
62}
