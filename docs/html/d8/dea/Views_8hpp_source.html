83template<Expression Source>
84class Reshape {
85public:

97 template<Integral... Indexes>
98 constexpr Reshape(Trait<Source>::Reference source, Indexes... indexes)
99 : shape_(indexes...)
100 , strides_(shape_)
101 , source_(source) {
102 std::size_t elements = 0;
103 for (std::size_t dimension = 0; dimension < sizeof...(indexes); ++dimension) {
104 elements += strides_[dimension] * (shape_[dimension] - 1);
105 }
106 elements += 1;
107 if (elements != std::accumulate(shape_.begin(), shape_.end(), 1ULL, std::multiplies<>{} ))
108 throw Exception("Shape mismatch: view must preserve total number of elements");
109 }
110
118 constexpr type dtype() const {
119 return source_.dtype();
120 }
121
128 constexpr Shape const& shape() const {
129 return shape_;
130 }
131
138 constexpr Strides const& strides() const {
139 return strides_;
140 }
141
151 std::ptrdiff_t offset() const {
152 return source_.offset();
153 }
154
155 Tensor forward() const;
156
157private:
158 Shape shape_;
159 Strides strides_;
160 typename Trait<Source>::Reference source_;
161};

198class Transpose {
199public:
200
207 constexpr Transpose(typename Trait<Source>::Reference source, std::pair<int, int> dimensions)
208 : shape_(source.shape())
209 , strides_(source.strides())
210 , source_(source)
211 , dimensions_(dimensions) {
212 auto rank = source.shape().rank();
213 std::swap(shape_[indexing::normalize(dimensions.first, rank)], shape_[indexing::normalize(dimensions.second, rank)]);
214 std::swap(strides_[indexing::normalize(dimensions.first, rank)], strides_[indexing::normalize(dimensions.second, rank)]);
215 }

257 }

259 Tensor forward() const;

261private:
262 Shape shape_;
263 Strides strides_;
264 typename Trait<Source>::Reference source_;
