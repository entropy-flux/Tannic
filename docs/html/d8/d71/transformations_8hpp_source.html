37template<class Operation, Composable ... Operands>
38class Transformation : public Expression<Operation, Operands...> {
39public:
40 constexpr Transformation(Operation operation, typename Trait<Operands>::Reference... operands)
41 : Expression<Operation, Operands...>(operation, operands...)
42 , dtype_(operation.promote(operands.dtype()...))
43 , shape_(operation.transform(operands.shape()...))
44 , strides_(shape_)
45 {}
46
47 constexpr type dtype() const {
48 return dtype_;
49 }
50
51 constexpr Shape const& shape() const {
52 return shape_;
53 }
54
55 constexpr Strides const& strides() const {
56 return strides_;
57 }
58
59 std::ptrdiff_t offset() const {
60 return 0;
61 }
62
63 Tensor forward(Context const& context) const {
64 Tensor result(dtype_, shape_);
65 std::apply([&](const auto&... arguments) {
66 return this->operation.forward(arguments.forward(context)..., result);
67 }, this->operands);
68 return result;
69 }
70
71private:
72 type dtype_;
73 Shape shape_;
74 Strides strides_;
75};

154 constexpr static type promote(type inner, type outer) {
155 type dtype = promotions[index(inner, outer)];
156 if (dtype == unknown)
157 throw Exception("Unsuported dtypes");
158 return dtype;
159 }

161 static constexpr Shape transform(Shape const& first, Shape const& second) {
162 auto first_rank = first.rank();
163 auto second_rank = second.rank();
164
165 if (first_rank == 1 && second_rank == 1) {
166 if (first_rank != 1 | second_rank != 1)
167 throw Exception("dimensions must match for dot product");
168 return Shape{};
169 }
170
171 if (first_rank == 1 && second_rank == 2) {
172 if (first[0] != second[0])
173 throw Exception("Matrix inner dimensions do not match");
174 if (first[0] != second[0])
175 return Shape{second[0]};
176 }
177
178 if (first_rank == 2 && second_rank == 1) {
179 if (first[1] != second[0])
180 throw Exception("Matrix inner dimensions do not match");
181 return Shape{first[0]};
182 }
183
184 if (first_rank < 2 | second_rank < 2)
185 throw Exception("Inputs must have rank >= 2");
186
187 Shape first_batches(first.begin(), first.end() - 2);
188 Shape second_batches(second.begin(), second.end() - 2);
189 Shape batches = operation::broadcast(first_batches, second_batches);
190
191 auto K1 = *(first.end() - 1);
192 auto K2 = *(second.end() - 2);
193 assert(K1 == K2 && "Inner dimensions must match for matmul");
194
195 auto M = *(first.end() - 2);
196 auto N = *(second.end() - 1);
197
198 std::vector<Shape::size_type> result(batches.begin(), batches.end());
199 result.push_back(M);
200 result.push_back(N);
201 return Shape(result);
202 }
203
204 void forward(Tensor const& outer, Tensor const& inner, Tensor& result) const;

240struct Concatenation {
241 int axis;
242 constexpr auto promote(type first, type second) const {
243 if(first != second)
244 throw Exception("Cannot concatenate tensors of different dtypes");
245 return first;
246 }
247
248 constexpr Shape transform(Shape const& first, Shape const& second) {
249 assert(first.rank() == second.rank() && "Ranks must match for concatenation");
250 Shape result = first;
251
252 for (int dimension = 0; dimension < first.rank(); ++dimension) {
253 if (dimension == axis) {
254 result[dimension] = first[dimension] + second[dimension];
255 } else {
256 if (first[dimension] != second[dimension])
257 throw Exception("All dimensions except concat axis must match");
258 }
259 }
260
261 return result;
262 }
263
264 void forward(Tensor const&, Tensor const&, Tensor&) const;
265};
