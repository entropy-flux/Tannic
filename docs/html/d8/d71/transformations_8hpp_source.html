57class Transformation : public Expression<Operation, Operands...> {
58public:
59 constexpr Transformation(Operation operation, typename Trait<Operands>::Reference... operands)
60 : Expression<Operation, Operands...>(operation, operands...)
61 , dtype_(operation.promote(operands.dtype()...))
62 , shape_(operation.transform(operands.shape()...))
63 , strides_(shape_)
64 {}

67 return dtype_;
68 }

75 return strides_;

78 std::ptrdiff_t offset() const {
79 return 0;

83 Tensor result(dtype_, shape_);

88 }

91 type dtype_;
92 Shape shape_;
93 Strides strides_;
94};

204 constexpr static type promote(type inner, type outer) {

225 static constexpr Shape transform(Shape const& first, Shape const& second) {
226 auto first_rank = first.rank();
227 auto second_rank = second.rank();
228
229 if (first_rank == 1 && second_rank == 1) {
230 if (first_rank != 1 | second_rank != 1)
231 throw Exception("dimensions must match for dot product");
232 return Shape{}; // Scalar result
233 }
234
235 if (first_rank == 1 && second_rank == 2) {
236 if (first[0] != second[0])
237 throw Exception("Matrix inner dimensions do not match");
238 if (first[0] != second[0])
239 return Shape{second[0]}; // Vector result
240 }
241
242 // Vector-matrix multiplication
243 if (first_rank == 2 && second_rank == 1) {
244 if (first[1] != second[0])
245 throw Exception("Matrix inner dimensions do not match");
246 return Shape{first[0]}; // Vector result
247 }
248
249 if (first_rank < 2 | second_rank < 2)
250 throw Exception("Inputs must have rank >= 2");
251
252 // Handle batch dimensions
253 Shape first_batches(first.begin(), first.end() - 2);
254 Shape second_batches(second.begin(), second.end() - 2);
255 Shape batches = operation::broadcast(first_batches, second_batches);
256
257 // Check matrix inner dimensions
258 auto K1 = *(first.end() - 1);
259 auto K2 = *(second.end() - 2);
260 assert(K1 == K2 && "Inner dimensions must match for matmul");
261
262 // Get output matrix dimensions
263 auto M = *(first.end() - 2);
264 auto N = *(second.end() - 1);
265
266 // Combine batch and matrix dimensions
267 std::vector<Shape::size_type> result(batches.begin(), batches.end());
268 result.push_back(M);
269 result.push_back(N);
270 return Shape(result);
271 }

273 void forward(Tensor const& outer, Tensor const& inner, Tensor& result) const;

362struct Concatenation {
363 int axis;
373 constexpr auto promote(type first, type second) const {
374 if(first != second)
375 throw Exception("Cannot concatenate tensors of different dtypes");
376 return first;
377 }
378
389 constexpr Shape transform(Shape const& first, Shape const& second) {
390 assert(first.rank() == second.rank() && "Ranks must match for concatenation");
391 Shape result = first;
392
393 for (int dimension = 0; dimension < first.rank(); ++dimension) {
394 if (dimension == axis) {
395 result[dimension] = first[dimension] + second[dimension];
396 } else {
397 if (first[dimension] != second[dimension])
398 throw Exception("All dimensions except concat axis must match");
399 }
400 }
401
402 return result;
403 }
404
405 void forward(Tensor const&, Tensor const&, Tensor&) const;
406};
