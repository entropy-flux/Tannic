60 : Expression<Operation, Operands...>(operation, operands...)
61 , dtype_(operation.promote(operands.dtype()...))

63 , strides_(shape_)
64 {}

79 return 0;

218 constexpr static type promote(type inner, type outer) {
219 type dtype = promotions[index(inner, outer)];
220 if (dtype == unknown)
221 throw Exception("Unsuported dtypes");

239 static constexpr Shape transform(Shape const& first, Shape const& second) {
240 auto first_rank = first.rank();
241 auto second_rank = second.rank();
242
243 if (first_rank == 1 && second_rank == 1) {
244 if (first_rank != 1 | second_rank != 1)
245 throw Exception("dimensions must match for dot product");
246 return Shape{}; // Scalar result
247 }
248
249 if (first_rank == 1 && second_rank == 2) {
250 if (first[0] != second[0])
251 throw Exception("Matrix inner dimensions do not match");
252 if (first[0] != second[0])
253 return Shape{second[0]}; // Vector result
254 }
255
256 // Vector-matrix multiplication
257 if (first_rank == 2 && second_rank == 1) {
258 if (first[1] != second[0])
259 throw Exception("Matrix inner dimensions do not match");
260 return Shape{first[0]}; // Vector result
261 }
262
263 if (first_rank < 2 | second_rank < 2)
264 throw Exception("Inputs must have rank >= 2");
265
266 // Handle batch dimensions
267 Shape first_batches(first.begin(), first.end() - 2);
268 Shape second_batches(second.begin(), second.end() - 2);
269 Shape batches = operation::broadcast(first_batches, second_batches);
270
271 // Check matrix inner dimensions
272 auto K1 = *(first.end() - 1);
273 auto K2 = *(second.end() - 2);
274 assert(K1 == K2 && "Inner dimensions must match for matmul");
275
276 // Get output matrix dimensions
277 auto M = *(first.end() - 2);
278 auto N = *(second.end() - 1);
279
280 // Combine batch and matrix dimensions
281 std::vector<Shape::size_type> result(batches.begin(), batches.end());
282 result.push_back(M);
283 result.push_back(N);
284 return Shape(result);
285 }

287 void forward(Tensor const& outer, Tensor const& inner, Tensor& result) const;

376struct Concatenation {
377 int axis;
387 constexpr auto promote(type first, type second) const {
388 if(first != second)
389 throw Exception("Cannot concatenate tensors of different dtypes");
390 return first;
391 }
392
403 constexpr Shape transform(Shape const& first, Shape const& second) {
404 assert(first.rank() == second.rank() && "Ranks must match for concatenation");
405 Shape result = first;
406
407 for (int dimension = 0; dimension < first.rank(); ++dimension) {
408 if (dimension == axis) {
409 result[dimension] = first[dimension] + second[dimension];
410 } else {
411 if (first[dimension] != second[dimension])
412 throw Exception("All dimensions except concat axis must match");
413 }
414 }
415
416 return result;
417 }
418
419 void forward(Tensor const&, Tensor const&, Tensor&) const;
420};
