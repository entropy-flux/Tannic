69template <Composable Source, class... Indexes>
70class Slice {
71public:
72
73 constexpr Slice(typename Trait<Source>::Reference source, std::tuple<Indexes...> indexes)
74 : dtype_(source.dtype())
75 , source_(source)
76 , indexes_(indexes)
77 {
78 if (source.rank() == 0) {
79 shape_ = Shape{};
80 strides_ = Strides{};
81 offset_ = 0;
82 }
83
84 else {
85 std::array<Shape::size_type, Shape::limit> shape{};
86 std::array<Strides::size_type, Strides::limit> strides{};
87 Shape::rank_type dimension = 0;
88 Shape::rank_type rank = 0;
89 offset_ = 0;
90 auto process = [&](const auto& argument) {
91 using Argument = std::decay_t<decltype(argument)>;
92 if constexpr (std::is_same_v<Argument, indexing::Range>) {
93 auto range = normalize(argument, source.shape()[dimension]);
94 auto size = range.stop - range.start;
95 shape[rank] = size;
96 strides[rank] = source.strides()[dimension];
97 offset_ += range.start * source.strides()[dimension] * dsizeof(dtype_);
98 rank++; dimension++;
99 }
100
101 else if constexpr (std::is_integral_v<Argument>) {
102 auto index = indexing::normalize(argument, source.shape()[dimension]);
103 offset_ += index * source.strides()[dimension] * dsizeof(dtype_);
104 dimension++;
105 }
106
107 else {
108 throw Exception("Unknown index type");
109 }
110 };
111
112 std::apply([&](const auto&... arguments) {
113 (process(arguments), ...);
114 }, indexes);
115
116 while (dimension < source.rank()) {
117 shape[rank] = source.shape()[dimension];
118 strides[rank] = source.strides()[dimension];
119 rank++; dimension++;
120 }
121
122 shape_ = Shape(shape.begin(), shape.begin() + rank);
123 strides_ = Strides(strides.begin(), strides.begin() + rank);
124 }
125 }

199template <Composable Source, class... Indexes>
200template <Arithmetic T>
201void Slice<Source, Indexes...>::operator=(T value) {
202 auto copy = [this](std::byte const* value, std::ptrdiff_t offset) {
203 if(rank() == 0) {
204 assign(value, offset);
205 return;
206 }
207 std::vector<std::size_t> indexes(rank(), 0);
208 bool done = false;
209
210 while (!done) {
211 std::size_t position = offset;
212 for (auto dimension = 0; dimension < rank(); ++dimension) {
213 position += indexes[dimension] * strides_[dimension] * dsizeof(dtype_);
214 }
215
216 assign(value, position);
217 done = true;
218 for (int dimension = rank() - 1; dimension >= 0; --dimension) {
219 if (++indexes[dimension] < shape_[dimension]) {
220 done = false;
221 break;
222 }
223 indexes[dimension] = 0;
224 }
225 }
226 };
227
228 switch (dtype_) {
229 case int8: { copy(tobytes(int8_t(value)), offset()); break; }
230 case int16: { copy(tobytes(int16_t(value)), offset()); break; }
231 case int32: { copy(tobytes(int32_t(value)), offset()); break; }
232 case int64: { copy(tobytes(int64_t(value)), offset()); break; }
233 case float16: { copy(tobytes(float16_t(value)), offset()); break; }
234 case bfloat16: { copy(tobytes(bfloat16_t(value)), offset()); break; }
235 case float32:{ copy(tobytes(float(value)), offset()); break; }
236 case float64:{ copy(tobytes(double(value)), offset()); break; }
237 default: throw Exception("Unsupported dtype for assignment");
238 }
239}

241template <Composable Source, class... Indexes>
242template <Arithmetic T>
243bool Slice<Source, Indexes...>::operator==(T value) const {
244 if (rank() != 0)
245 throw Exception("Cannot compare an scalar to a non scalar slice");
246
247 switch (dtype_) {
248 case int8: { int8_t casted = value; return compare(tobytes(casted), offset()); }
249 case int16: { int16_t casted = value; return compare(tobytes(casted), offset()); }
250 case int32: { int32_t casted = value; return compare(tobytes(casted), offset()); }
251 case int64: { int64_t casted = value; return compare(tobytes(casted), offset()); }
252 case float16: { float16_t half = value; return compare(tobytes(half) , offset()); }
253 case bfloat16: { bfloat16_t half = value; return compare(tobytes(half) , offset()); }
254 case float32: { float casted = value; return compare(tobytes(casted), offset()); }
255 case float64: { double casted = value; return compare(tobytes(casted), offset()); }
256 default: throw Exception("Unsupported dtype for comparison");
257 }
