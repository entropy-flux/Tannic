38class Convolution1D {
39public:
40 std::array<std::size_t, 1> strides;
41 std::array<std::size_t, 1> padding;
42
43 constexpr type promote(type signal, type kernel) const {
44 if (signal != kernel)
45 throw Exception("Dtypes must match in convolutions.");
46 return signal;
47 }
48
49 constexpr Shape transform(Shape const& signal, Shape const& kernel) const {
50 if (signal.rank() != 3 || kernel.rank() != 3)
51 throw Exception("Only rank 3 tensors supported for Conv1D.");
52
53 std::size_t N = signal[0]; // Batch size
54 std::size_t C_in = signal[1]; // Input channels
55 std::size_t L_in = signal[2]; // Input length
56
57 std::size_t C_out = kernel[0]; // Output channels
58 std::size_t K_in = kernel[1]; // Kernel input channels
59 std::size_t K_len = kernel[2]; // Kernel length
60
61 if (C_in != K_in)
62 throw Exception("Input channels must match kernel channels.");
63
64 std::size_t L_out = (L_in + 2 * padding[0] - K_len) / strides[0] + 1;
65 return Shape{N, C_out, L_out};
66 }
67
68 void forward(Tensor const& signal, Tensor const& kernel, Tensor& result) const;
69};

71class Convolution2D {
72public:
73 std::array<std::size_t, 2> strides;
74 std::array<std::size_t, 2> padding;
75
76 constexpr type promote(type signal, type kernel) const {
77 if (signal != kernel)
78 throw Exception("Dtypes must match in convolutions.");
79 return signal;
80 }
81
82 constexpr type promote(type signal, type kernel, type bias) const {
83 if (signal != kernel || signal != bias)
84 throw Exception("Signal, kernel, and bias dtypes must all match.");
85 return signal;
86 }
87
88 constexpr Shape transform(Shape const& signal, Shape const& kernel) const {
89 if (signal.rank() != 4 | kernel.rank() != 4)
90 throw Exception("Only rank 4 tensors supported.");
91
92 std::size_t N = signal[0]; // Batch size
93 std::size_t C_in = signal[1]; // Input channels
94 std::size_t H_in = signal[2]; // Input height
95 std::size_t W_in = signal[3]; // Input width
96
97 std::size_t C_out = kernel[0]; // Output channels
98 std::size_t K_in = kernel[1]; // Kernel input channels
99 std::size_t K_h = kernel[2]; // Kernel height
100 std::size_t K_w = kernel[3]; // Kernel width
101
102 if (C_in != K_in)
103 throw Exception("Input channels must match kernel channels.");
104
105 std::size_t H_out = (H_in + 2 * padding[0] - K_h) / strides[0] + 1;
106 std::size_t W_out = (W_in + 2 * padding[1] - K_w) / strides[1] + 1;
107
108 return Shape{N, C_out, H_out, W_out};
109 }
110
111 constexpr Shape transform(Shape const& signal, Shape const& kernel, Shape const& bias) const {
112 Shape out = transform(signal, kernel);
113 if (bias.rank() != 1 || bias[0] != out[1])
114 throw Exception("Bias shape must be [C_out].");
115 return out;
116 }
117
118 void forward(Tensor const& signal, Tensor const& kernel, Tensor& result) const;
119 void forward(Tensor const& signal, Tensor const& kernel, Tensor const& bias, Tensor& result) const;
120};

135template<Composable Signal, Composable Kernel>
136constexpr auto convolve1D(Signal&& signal, Kernel&& kernel, std::size_t stride, std::size_t padding) {
137 if (stride == 0)
138 throw Exception("Stride must be non-zero for Conv1D.");
139
140 return Transformation<Convolution1D, Signal, Kernel>(
141 {{stride}, {padding}},
142 std::forward<Signal>(signal),
143 std::forward<Kernel>(kernel)
144 );
145}
