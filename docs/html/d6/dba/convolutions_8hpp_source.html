57class Convolution1D {
58public:
59 std::array<std::size_t, 1> strides;
60 std::array<std::size_t, 1> padding;
61
72 constexpr type promote(type signal, type kernel) const {
73 if (signal != kernel)
74 throw Exception("Dtypes must match in convolutions.");
75 return signal;

77
96 constexpr Shape transform(Shape const& signal, Shape const& kernel) const {
97 if (signal.rank() != 3 || kernel.rank() != 3)
98 throw Exception("Only rank 3 tensors supported for Conv1D.");
99
100 std::size_t N = signal[0]; // Batch size
101 std::size_t C_in = signal[1]; // Input channels
102 std::size_t L_in = signal[2]; // Input length
103
104 std::size_t C_out = kernel[0]; // Output channels
105 std::size_t K_in = kernel[1]; // Kernel input channels
106 std::size_t K_len = kernel[2]; // Kernel length
107
108 if (C_in != K_in)
109 throw Exception("Input channels must match kernel channels.");
110
111 std::size_t L_out = (L_in + 2 * padding[0] - K_len) / strides[0] + 1;
112 return Shape{N, C_out, L_out};
113 }
114
122 void forward(Tensor const& signal, Tensor const& kernel, Tensor& result) const;
123};

133class Convolution2D {
134public:
135 std::array<std::size_t, 2> strides;
136 std::array<std::size_t, 2> padding;
137
148 constexpr type promote(type signal, type kernel) const {
149 if (signal != kernel)
150 throw Exception("Dtypes must match in convolutions.");
151 return signal;
152 }
153
154 constexpr type promote(type signal, type kernel, type bias) const {
155 if (signal != kernel || signal != bias)
156 throw Exception("Signal, kernel, and bias dtypes must all match.");
157 return signal;
158 }
159
160
181 constexpr Shape transform(Shape const& signal, Shape const& kernel) const {
182 if (signal.rank() != 4 | kernel.rank() != 4)
183 throw Exception("Only rank 4 tensors supported.");
184
185 std::size_t N = signal[0]; // Batch size
186 std::size_t C_in = signal[1]; // Input channels
187 std::size_t H_in = signal[2]; // Input height
188 std::size_t W_in = signal[3]; // Input width
189
190 std::size_t C_out = kernel[0]; // Output channels
191 std::size_t K_in = kernel[1]; // Kernel input channels
192 std::size_t K_h = kernel[2]; // Kernel height
193 std::size_t K_w = kernel[3]; // Kernel width
194
195 if (C_in != K_in)
196 throw Exception("Input channels must match kernel channels.");
197
198 std::size_t H_out = (H_in + 2 * padding[0] - K_h) / strides[0] + 1;
199 std::size_t W_out = (W_in + 2 * padding[1] - K_w) / strides[1] + 1;
200
201 return Shape{N, C_out, H_out, W_out};
202 }
203
204 constexpr Shape transform(Shape const& signal, Shape const& kernel, Shape const& bias) const {
205 Shape out = transform(signal, kernel);
206 if (bias.rank() != 1 || bias[0] != out[1])
207 throw Exception("Bias shape must be [C_out].");
208 return out;
209 }
210
218 void forward(Tensor const& signal, Tensor const& kernel, Tensor& result) const;
219 void forward(Tensor const& signal, Tensor const& kernel, Tensor const& bias, Tensor& result) const;
220};

234template<Composable Signal, Composable Kernel>
235constexpr auto convolve1D(Signal&& signal, Kernel&& kernel, std::size_t stride, std::size_t padding) {
236 if (stride == 0)
237 throw Exception("Stride must be non-zero for Conv1D.");
238
239 return Transformation<Convolution1D, Signal, Kernel>(
240 {{stride}, {padding}},
241 std::forward<Signal>(signal),
242 std::forward<Kernel>(kernel)
