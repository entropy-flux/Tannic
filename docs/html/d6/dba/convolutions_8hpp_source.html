102 constexpr Shape transform(Shape const& signal, Shape const& kernel) const {
103 if (signal.rank() != 3 || kernel.rank() != 3)
104 throw Exception("Only rank 3 tensors supported for Conv1D.");
105
106 std::size_t N = signal[0]; // Batch size
107 std::size_t C_in = signal[1]; // Input channels
108 std::size_t L_in = signal[2]; // Input length
109
110 std::size_t C_out = kernel[0]; // Output channels
111 std::size_t K_in = kernel[1]; // Kernel input channels
112 std::size_t K_len = kernel[2]; // Kernel length
113
114 if (C_in != K_in)
115 throw Exception("Input channels must match kernel channels.");
116
117 std::size_t L_out = (L_in + 2 * padding[0] - K_len) / strides[0] + 1;
118 return Shape{N, C_out, L_out};

229template<auto Dimension, Expression Signal, Expression Kernel>
230constexpr auto convolve(Signal&& signal, Kernel&& kernel, std::size_t strides, std::size_t padding) {
231 if (strides == 0)
232 throw Exception("Strides should be non zero.");
233
234 return Transformation<Convolution<Dimension>, Signal, Kernel>(
235 {{strides}, {padding}},
236 std::forward<Signal>(signal),
237 std::forward<Kernel>(kernel)
238 );
239}

254template<auto Dimension, Expression Signal, Expression Kernel>
255constexpr auto convolve(Signal&& signal, Kernel&& kernel, std::array<std::size_t, Dimension> strides, std::array<std::size_t, Dimension> padding) {
256 for (std::size_t i = 0; i < Dimension; ++i) {
257 if (strides[i] == 0)
258 throw Exception("Strides should be non zero.");
259 }
260
261 return Transformation<Convolution<Dimension>, Signal, Kernel>(
262 {strides, padding},
263 std::forward<Signal>(signal),
264 std::forward<Kernel>(kernel)
265 );
266}
