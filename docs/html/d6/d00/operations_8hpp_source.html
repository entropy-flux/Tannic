208static constexpr Shape broadcast(Shape const& first, Shape const& second) {
209 auto first_rank = first.rank();
210 auto second_rank = second.rank();
211 auto rank = std::max(first_rank, second_rank);
212 Shape shape;
213 for (auto dimension = 0; dimension < rank; ++dimension) {
214 auto first_dimension = (dimension < rank - first_rank) ? 1 : first[dimension - (rank - first_rank)];
215 auto second_dimension = (dimension < rank - second_rank) ? 1 : second[dimension - (rank - second_rank)];
216
217 if (!(first_dimension == second_dimension || first_dimension == 1 || second_dimension == 1)) {
218 throw Exception("Shapes are not broadcast-compatible.");
219 }
220 shape.expand(std::max(first_dimension, second_dimension));
221 }
222 return shape;
223}
