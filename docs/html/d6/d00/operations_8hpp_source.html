133static constexpr inline type promote(type first, type second) {
134 return static_cast<int>(first) > static_cast<int>(second) ? first : second;
135}
136
137static constexpr Shape broadcast(Shape const& first, Shape const& second) {
138 auto first_rank = first.rank();
139 auto second_rank = second.rank();
140 auto rank = std::max(first_rank, second_rank);
141 Shape shape;
142 for (auto dimension = 0; dimension < rank; ++dimension) {
143 auto first_dimension = (dimension < rank - first_rank) ? 1 : first[dimension - (rank - first_rank)];
144 auto second_dimension = (dimension < rank - second_rank) ? 1 : second[dimension - (rank - second_rank)];
145
146 if (!(first_dimension == second_dimension || first_dimension == 1 || second_dimension == 1)) {
147 throw Exception("Shapes are not broadcast-compatible.");
148 }
149 shape.expand(std::max(first_dimension, second_dimension));
150 }
151 return shape;
152}
