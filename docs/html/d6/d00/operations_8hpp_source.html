152 return static_cast<int>(first) > static_cast<int>(second) ? first : second;

176static constexpr Shape broadcast(Shape const& first, Shape const& second) {
177 auto first_rank = first.rank();
178 auto second_rank = second.rank();
179 auto rank = std::max(first_rank, second_rank);
180 Shape shape;
181 for (auto dimension = 0; dimension < rank; ++dimension) {
182 auto first_dimension = (dimension < rank - first_rank) ? 1 : first[dimension - (rank - first_rank)];
183 auto second_dimension = (dimension < rank - second_rank) ? 1 : second[dimension - (rank - second_rank)];
184
185 if (!(first_dimension == second_dimension || first_dimension == 1 || second_dimension == 1)) {
186 throw Exception("Shapes are not broadcast-compatible.");
187 }
188 shape.expand(std::max(first_dimension, second_dimension));
189 }
190 return shape;
191}
