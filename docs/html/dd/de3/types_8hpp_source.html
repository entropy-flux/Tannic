251template <typename T>
252constexpr inline type dtypeof() {
253 if constexpr (std::is_same_v<T, bool>) return boolean;
254 else if constexpr (std::is_same_v<T, int8_t>) return int8;
255 else if constexpr (std::is_same_v<T, int16_t>) return int16;
256 else if constexpr (std::is_same_v<T, int32_t>) return int32;
257 else if constexpr (std::is_same_v<T, int64_t>) return int64;
258 else if constexpr (std::is_same_v<T, float16_t>) return float16;
259 else if constexpr (std::is_same_v<T, bfloat16_t>) return bfloat16;
260 else if constexpr (std::is_same_v<T, float>) return float32;
261 else if constexpr (std::is_same_v<T, double>) return float64;
262 else if constexpr (std::is_same_v<T, std::complex<float>>) return complex64;
263 else if constexpr (std::is_same_v<T, std::complex<double>>) return complex128;
264 else return unknown;
265}
