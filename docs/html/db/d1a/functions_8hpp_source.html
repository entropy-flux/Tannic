28template<Functional Functor, Composable Argument>
29class Function : public Expression<Functor, Argument> {
30public:
31 constexpr Function(Functor functor, typename Trait<Argument>::Reference argument)
32 : Expression<Functor, Argument>(functor, argument)
33 , strides_(argument.shape())
34 {}
35
36 constexpr type dtype() const {
37 return std::get<0>(this->operands).dtype();
38 }
39
40 constexpr Shape const& shape() const {
41 return std::get<0>(this->operands).shape();
42 }
43
44 constexpr Strides const& strides() const {
45 return strides_;
46 }
47
48 auto offset() const {
49 return 0;
50 }
51
52 Tensor forward(Context const& context) const {
53 Tensor source = std::get<0>(this->operands).forward(context);
54 Tensor target(dtype(), shape());
55 this->operation(source, target);
56 return target;
57 }
58
59private:
60 Strides strides_;
61};
