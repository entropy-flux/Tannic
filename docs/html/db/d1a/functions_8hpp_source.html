65template<Functional Functor, Composable Argument>
66class Function : public Expression<Functor, Argument> {
67public:
68 constexpr Function(Functor functor, typename Trait<Argument>::Reference argument)
69 : Expression<Functor, Argument>(functor, argument)
70 {}
71
72 constexpr type dtype() const {
73 return std::get<0>(this->operands).dtype();
74 }
75
76 constexpr Shape const& shape() const {
77 return std::get<0>(this->operands).shape();
78 }
79
80 constexpr Strides const& strides() const {
81 return std::get<0>(this->operands).strides();
82 }
83
84 auto offset() const {
85 return std::get<0>(this->operands).offset();
86 }
87
88 Tensor forward() const {
89 Tensor source = std::get<0>(this->operands).forward();
90 Tensor target(dtype(), shape(), strides(), offset());
91 this->operation(source, target);
92 return target;
93 }
94};
