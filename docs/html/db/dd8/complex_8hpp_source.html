46template<class Coordinates, Composable Source>
47class Complexification<Coordinates, Source> {
48public:
49 typename Trait<Source>::Reference source;
50
51 constexpr Complexification(Trait<Source>::Reference source)
52 : source(source) {
53 switch (source.dtype()) {
54 case float32: dtype_ = complex64; break;
55 case float64: dtype_ = complex128; break;
56 default:
57 throw Exception("Complex view error: source tensor dtype must be float32 or float64");
58 }
59
60 if (source.shape().back() != 2) {
61 throw Exception("Complex view error: last dimension must be size 2 (real + imag).");
62 }
63
64 shape_ = Shape(source.shape().begin(), source.shape().end() - 1);
65 strides_ = Strides(source.strides().begin(), source.strides().end() - 1);
66
67 for (int dimension = 0; dimension < strides_.rank(); ++dimension) {
68 strides_[dimension] /= 2;
69 }
70
71 strides_[-1] = 1;
72 }
73
74 constexpr type dtype() const {
75 return dtype_;
76 }
77
78 constexpr Shape const& shape() const {
79 return shape_;
80 }
81
82 constexpr Strides const& strides() const {
83 return strides_;
84 }
85
86 std::ptrdiff_t offset() const {
87 return source.offset();
88 }
89
90 Tensor forward(Context const& context) const;
91
92private:
93 type dtype_;
94 Shape shape_;
95 Strides strides_;
96};

98template<class Coordinates, Composable Real, Composable Imaginary>
99class Complexification<Coordinates, Real, Imaginary> {
100public:
101 typename Trait<Real>::Reference real;
102 typename Trait<Imaginary>::Reference imaginary;
103
104 constexpr Complexification(typename Trait<Real>::Reference real, typename Trait<Imaginary>::Reference imaginary)
105 : real(real)
106 , imaginary(imaginary)
107 {
108 if (real.shape() != imaginary.shape() | real.strides() != imaginary.strides())
109 throw Exception("Complexification error: real and imaginary part layouts must match");
110
111 if (real.dtype() == float64 || imaginary.dtype() == float64) {
112 dtype_ = complex128;
113 }
114
115 else {
116 dtype_ = complex64;
117 }
118 }
119
120 constexpr type dtype() const {
121 return dtype_;
122 }
123
124 constexpr Shape const& shape() const {
125 return real.shape();
126 }
127
128 constexpr Strides const& strides() const {
129 return real.strides();
130 }
131
132 std::ptrdiff_t offset() const {
133 return 0;
134 }
135
136 Tensor forward(Context const& context) const;
137
138private:
139 type dtype_;
140};

142template<Composable Source>
143class Realification {
144public:
145 typename Trait<Source>::Reference source;
146
147 constexpr Realification(Trait<Source>::Reference source)
148 : source(source) {
149 switch (source.dtype()) {
150 case complex64: dtype_ = float32; break;
151 case complex128: dtype_ = float64; break;
152 default:
153 throw Exception("Real view error: source tensor dtype must be complex64 or complex128");
154 }
155
156 if (source.strides()[-1] != 1) {
157 throw Exception("Real view error: source tensor is not in interleaved real/imag format");
158 }
159
160 shape_ = Shape(source.shape().begin(), source.shape().end());
161 shape_.expand(2);
162
163 strides_ = Strides(source.strides().begin(), source.strides().end());
164 strides_.expand(1);
165 for (auto dimension = 0; dimension < strides_.rank() - 1; ++dimension) {
166 strides_[dimension] *= 2;
167 }
168 }
169
170 constexpr type dtype() const {
171 return dtype_;
172 }
173
174 constexpr Shape const& shape() const {
175 return shape_;
176 }
177
178 constexpr Strides const& strides() const {
179 return strides_;
180 }
181
182 std::ptrdiff_t offset() const {
183 return source.offset();
184 }
185
186 Tensor forward(Context const& context) const;
187
188private:
189 type dtype_;
190 Shape shape_;
191 Strides strides_;
192};
