34template<class Reducer, Expression Operand>
35class Reduction {
36public:
37 Reducer reducer;
38 typename Trait<Operand>::Reference operand;
39
40 constexpr Reduction(Reducer reducer, typename Trait<Operand>::Reference operand)
41 : reducer(reducer)
42 , operand(operand)
43 , dtype_(reducer.reduce(operand.dtype()))
44 , shape_(reducer.reduce(operand.shape()))
45 , strides_(reducer.reduce(operand.strides())) {}
46
47 constexpr type dtype() const {
48 return dtype_;
49 }
50
51 constexpr Shape const& shape() const {
52 return shape_;
53 }
54
55 constexpr Strides const& strides() const {
56 return strides_;
57 }
58
59 std::ptrdiff_t offset() const {
60 return 0;
61 }
62
63 Tensor forward() const {
64 Tensor source = source.forward();
65 Tensor result(dtype_, shape_, strides_, offset());
66 reducer.forward(source, result);
67 return result;
68 }

70private:
71 type dtype_;
72 Shape shape_;
73 Strides strides_;

75};

77struct Argmax {
78 int axis;
79 void forward(Tensor const&, Tensor&) const;
80
81 constexpr type reduce(type dtype) const {
82 return int64;
83 }
84
85 constexpr Shape reduce(Shape const& shape) const {
86 assert(shape.rank() >= 1 && "Argmax requires at least 1D tensor");
87 return Shape(shape.begin(), shape.end()-1);
88 }
89
90 constexpr Strides reduce(Strides const& strides) const {
91 if (strides.rank() == 1) {
92 return Strides();
93 }
94
95 uint8_t rank = strides.rank() - 1;
96 std::vector<std::size_t> sizes(rank);
97 for(uint8_t dimension = 0; dimension < rank - 2; ++dimension) {
98 sizes[dimension] = strides[dimension];
99 }
100 sizes[rank - 1] = strides.back();
101 return Strides(sizes.begin(), sizes.end());
102 }
103};

133template<Expression Source>
134constexpr auto argmax(Source&& source, int axis = -1) {
135 assert(axis == -1 && "Axis different from default not working in argmax. Open a PR if you find the issue.");
136 return Reduction<Argmax, Source>{
137 {indexing::normalize(axis, source.shape().rank())}, std::forward<Source>(source)
138 };
139}
