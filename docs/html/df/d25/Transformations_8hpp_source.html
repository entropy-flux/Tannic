124 static constexpr Shape broadcast(Shape const& first, Shape const& second) {
125 auto first_rank = first.rank();
126 auto second_rank = second.rank();
127
128 if (first_rank == 1 && second_rank == 1) {
129 assert(first[0] == second[0] && "Vector dimensions must match for dot product");
130 return Shape{};
131 }
132
133 if (first_rank == 1 && second_rank == 2) {
134 assert(first[0] == second[1] && "Matrix inner dimensions do not match");
135 return Shape{second[0]};
136 }
137
138 if (first_rank == 2 && second_rank == 1) {
139 assert(first[1] == second[0] && "Matrix inner dimensions do not match");
140 return Shape{first[0]};
141 }
142
143 assert(first_rank >= 2 && second_rank >= 2 && "Inputs must have rank >= 2");
144
145 Shape first_batches(first.begin(), first.end() - 2);
146 Shape second_batches(second.begin(), second.end() - 2);
147 Shape batches = operation::broadcast(first_batches, second_batches);
148
149 auto K1 = *(first.end() - 1);
150 auto K2 = *(second.end() - 2);
151 assert(K1 == K2 && "Inner dimensions must match for matmul");
152
153 auto M = *(first.end() - 2);
154 auto N = *(second.end() - 1);
155
156 std::vector<Shape::size_type> result(batches.begin(), batches.end());
157 result.push_back(M);
158 result.push_back(N);
159 return Shape(result);
160 }
161};
