53template<>
54class Convolution<1> {
55public:
56 std::array<std::size_t, 1> strides;
57 std::array<std::size_t, 1> padding;
58
59 constexpr type promote(type signal, type kernel) const {
60 if (signal != kernel)
61 throw Exception("Dtypes must match in convolutions.");
62 return signal;
63 }
64
65 constexpr Shape transform(Shape const& signal, Shape const& kernel) const {
66 if (signal.rank() != 3 || kernel.rank() != 3)
67 throw Exception("Only rank 3 tensors supported for Conv1D.");
68
69 std::size_t N = signal[0];
70 std::size_t C_in = signal[1];
71 std::size_t L_in = signal[2];
72
73 std::size_t C_out = kernel[0];
74 std::size_t K_in = kernel[1];
75 std::size_t K_len = kernel[2];
76
77 if (C_in != K_in)
78 throw Exception("Input channels must match kernel channels.");
79
80 std::size_t L_out = (L_in + 2 * padding[0] - K_len) / strides[0] + 1;
81 return Shape{N, C_out, L_out};
82 }
83
84 void forward(Tensor const&, Tensor const&, Tensor&) const;
85};

87template<>
88class Convolution<2> {
89public:
90 std::array<std::size_t, 2> strides;
91 std::array<std::size_t, 2> padding;
92
93 constexpr type promote(type signal, type kernel) const {
94 if (signal != kernel)
95 throw Exception("Dtypes must match in convolutions.");
96 return signal;
97 }
98
99 constexpr Shape transform(Shape const& signal, Shape const& kernel) const {
100 if (signal.rank() != 4 | kernel.rank() != 4)
101 throw Exception("Only rank 4 tensors supported.");
102 std::size_t N = signal[0];
103 std::size_t C_in = signal[1];
104 std::size_t H_in = signal[2];
105 std::size_t W_in = signal[3];
106
107 std::size_t C_out = kernel[0];
108 std::size_t K_in = kernel[1];
109 std::size_t K_h = kernel[2];
110 std::size_t K_w = kernel[3];
111
112 if (C_in != K_in)
113 throw Exception("Input channels must match kernel channels.");
114
115 std::size_t H_out = (H_in + 2 * padding[0] - K_h) / strides[0] + 1;
116 std::size_t W_out = (W_in + 2 * padding[1] - K_w) / strides[1] + 1;
117
118 return Shape{N, C_out, H_out, W_out};
119 }
120
121 void forward(Tensor const&, Tensor const&, Tensor&) const;
122};

125template<auto Dimension, Expression Signal, Expression Kernel>
126constexpr auto convolve(Signal&& signal, Kernel&& kernel, std::size_t strides, std::size_t padding) {
127 if (strides == 0 | strides == 0)
128 throw Exception("Strides should be non zero.");
129
130 return Transformation<Convolution<Dimension>, Signal, Kernel>(
131 {{strides}, {padding}},
132 std::forward<Signal>(signal),
133 std::forward<Kernel>(kernel)
134 );
135}

138template<auto Dimension, Expression Signal, Expression Kernel>
139constexpr auto convolve(Signal&& signal, Kernel&& kernel, std::array<std::size_t, Dimension> strides, std::array<std::size_t, Dimension> padding) {
140 if (strides[0] == 0 | strides[1] == 0)
141 throw Exception("Strides should be non zero.");
142
143 return Transformation<Convolution<Dimension>, Signal, Kernel>(
144 {strides, padding},
145 std::forward<Signal>(signal),
146 std::forward<Kernel>(kernel)
