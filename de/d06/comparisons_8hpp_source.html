28class Comparison {
29public:
30 Criteria criteria;

34 constexpr Comparison(Criteria criteria, typename Trait<First>::Reference first, typename Trait<Second>::Reference second)
35 : criteria(criteria)
36 , first(first)
37 , second(second)
38 , shape_(first.shape())
39 , strides_(shape_)
40 {
41 if(first.shape() != second.shape())
42 throw Exception("Cannot compare tensors of different shape");
43 }
44
45 constexpr type dtype() const {
46 return boolean;
47 }

49 constexpr Shape const& shape() const {
50 return shape_;
51 }

53 constexpr Strides const& strides() const {

55 }
56
57 constexpr std::ptrdiff_t offset() const {
58 return 0;
59 }

61 Tensor forward() const {

63 criteria.forward(first, second, result);

65 }
