83template<Expression Source>
84class View {
85public:
86
97 template<Integral... Sizes>
98 constexpr View(typename Trait<Source>::Reference source, Sizes... sizes)
99 : source_(source)
100 {
101 std::array<long long, sizeof...(Sizes)> requested{ static_cast<long long>(sizes)... };
102
103 std::size_t source_elements = std::accumulate(
104 source.shape().begin(), source.shape().end(), 1ULL, std::multiplies<>{}
105 );
106
107 // Step 3. Handle -1 (infer axis)
108 int infer_axis = -1;
109 std::size_t known_product = 1;
110 for (std::size_t i = 0; i < requested.size(); ++i) {
111 if (requested[i] == -1) {
112 if (infer_axis != -1)
113 throw Exception("Only one dimension can be inferred (-1) in view");
114 infer_axis = static_cast<int>(i);
115 } else if (requested[i] < 0) {
116 throw Exception("Invalid negative dimension in view");
117 } else {
118 known_product *= static_cast<std::size_t>(requested[i]);
119 }
120 }
121
122 if (infer_axis != -1) {
123 if (source_elements % known_product != 0)
124 throw Exception("Cannot infer dimension: source elements not divisible");
125 requested[infer_axis] = static_cast<long long>(source_elements / known_product);
126 }
127
128 for (auto v : requested) {
129 shape_.expand(static_cast<std::size_t>(v));
130 }
131
132 std::size_t new_elements = std::accumulate(shape_.begin(), shape_.end(), 1ULL, std::multiplies<>{});
133 if (new_elements != source_elements)
134 throw Exception("Shape mismatch: view must preserve total number of elements");
135
136 strides_ = Strides(shape_);
137 }
138

147 constexpr type dtype() const {
148 return source_.dtype();
149 }

158 return shape_;
159 }

168 return strides_;
169 }

181 return source_.offset();
182 }

184 Tensor forward() const;

186private:
187 Shape shape_;
188 Strides strides_;
189 typename Trait<Source>::Reference source_;
190};

397template<Expression Source>
398class Expansion {
399public:
410 template<Integral... Sizes>
411 constexpr Expansion(typename Trait<Source>::Reference source, Sizes... sizes)
412 : source_(source) {
413 std::array<long long, sizeof...(Sizes)> requested{ static_cast<long long>(sizes)... };
414 auto src_shape = source.shape();
415 auto src_rank = src_shape.rank();
416 auto dst_rank = requested.size();
417
418 // Rule: you can expand to a higher rank by prepending new dims
419 if (dst_rank < src_rank)
420 throw Exception("Expansion target rank must be >= source rank");
421
422 std::size_t src_offset = dst_rank - src_rank;
423 for (std::size_t i = 0; i < dst_rank; ++i) {
424 long long req = requested[i];
425 std::size_t target;
426
427 if (req == -1) {
428 if (i < src_offset) {
429 throw Exception("Cannot use -1 for new leading dimensions");
430 }
431 target = src_shape[i - src_offset];
432 } else if (req <= 0) {
433 throw Exception("Expansion size must be positive or -1");
434 } else {
435 target = static_cast<std::size_t>(req);
436 }
437
438 // Copy stride rules
439 if (i < src_offset) {
440 // new leading dimensions must be size > 0, stride 0
441 shape_.expand(target);
442 strides_.expand(0);
443 } else {
444 auto src_dim = src_shape[i - src_offset];
445 auto src_stride = source.strides()[i - src_offset];
446
447 if (src_dim == 1 && target > 1) {
448 shape_.expand(target);
449 strides_.expand(0); // broadcast
450 } else if (src_dim == target) {
451 shape_.expand(target);
452 strides_.expand(src_stride);
453 } else {
454 throw Exception("Expansion only allows -1 (keep) or broadcasting singleton dims");
455 }
456 }
457 }
458 }

626template<Expression Source>
627class Unsqueeze {
628public:
642 template<Integral... Axes>
643 constexpr Unsqueeze(typename Trait<Source>::Reference source, Axes... axes)
644 : source_(source) {
645 auto rank = source.shape().rank();
646 std::vector<std::size_t> normalized{ static_cast<std::size_t>(indexing::normalize(axes, rank + sizeof...(axes)))... };
647 std::sort(normalized.begin(), normalized.end());
648
649 size_t dimensions = rank + normalized.size();
650 size_t index = 0;
651 size_t axis = 0;
652
653 for (auto dimension = 0; dimension < dimensions; ++dimension) {
654 if (axis < normalized.size() && dimension == normalized[axis]) {
655 shape_.expand(1);
656 strides_.expand( (index < source.strides().rank()) ? source.strides()[index] : 1 );
657 ++axis;
658 } else {
659 shape_.expand(source.shape()[index]);
660 strides_.expand(source.strides()[index]);
661 ++index;
662 }
663 }
664 }

739template<Expression Source>
740class Flatten {
741public:
742 constexpr Flatten(typename Trait<Source>::Reference source, int start = 0, int end = -1)
743 : source_(source) {
744 auto rank = source.shape().rank();
745
746 start = indexing::normalize(start, rank);
747 end = indexing::normalize(end, rank);
748
749 if (start > end) {
750 throw Exception("Flatten requires start_dim <= end_dim");
751 }
752
753 for (int dimension = 0; dimension < start; ++dimension) {
754 shape_.expand(source.shape()[dimension]);
755 strides_.expand(source.strides()[dimension]);
756 }
757
758 std::size_t flattened = 1;
759 for (int dimension = start; dimension <= end; ++dimension) {
760 flattened *= source.shape()[dimension];
761 }
762 shape_.expand(flattened);
763 strides_.expand(source.strides()[end]);
764
765 for (int dimension = end + 1; dimension < rank; ++dimension) {
766 shape_.expand(source.shape()[dimension]);
767 strides_.expand(source.strides()[dimension]);
768 }
769 }
