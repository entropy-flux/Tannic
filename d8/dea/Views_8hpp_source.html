97 template<Integral... Sizes>
98 constexpr View(Trait<Source>::Reference source, Sizes... sizes)
99 : shape_(sizes...)
100 , strides_(shape_)
101 , source_(source) {
102 std::size_t elements = 0;
103 for (std::size_t dimension = 0; dimension < sizeof...(sizes); ++dimension) {
104 elements += strides_[dimension] * (shape_[dimension] - 1);
105 }
106 elements += 1;
107 if (elements != std::accumulate(shape_.begin(), shape_.end(), 1ULL, std::multiplies<>{} ))
108 throw Exception("Shape mismatch: view must preserve total number of elements");
109 }
110
118 constexpr type dtype() const {
119 return source_.dtype();
120 }
121
128 constexpr Shape const& shape() const {
129 return shape_;
130 }
131
138 constexpr Strides const& strides() const {
139 return strides_;
140 }

151 std::ptrdiff_t offset() const {
152 return source_.offset();
153 }

155 Tensor forward() const;

157private:
158 Shape shape_;
159 Strides strides_;
160 typename Trait<Source>::Reference source_;

184
191 constexpr Transpose(typename Trait<Source>::Reference source, std::pair<int, int> dimensions)
192 : shape_(source.shape())
193 , strides_(source.strides())
194 , source_(source)
195 , dimensions_(dimensions) {
196 auto rank = source.shape().rank();
197 std::swap(shape_[indexing::normalize(dimensions.first, rank)], shape_[indexing::normalize(dimensions.second, rank)]);
198 std::swap(strides_[indexing::normalize(dimensions.first, rank)], strides_[indexing::normalize(dimensions.second, rank)]);

273template<Expression Source, Integral ... Indexes>
274class Permutation {
275public:
287 constexpr Permutation(typename Trait<Source>::Reference source, std::tuple<Indexes...> indexes)
288 : source_(source)
289 {
290 if (sizeof...(Indexes) != source_.shape().rank()) {
291 throw Exception("Permutation rank must equal tensor rank");
292 }
293
294 std::apply([&](auto... indexes) {
295 (([&]{
296 int dimension = indexing::normalize(indexes, source_.shape().rank());
297 shape_.expand(source_.shape()[dimension]);
298 strides_.expand(source_.strides()[dimension]);
299 }()), ...);
300 }, indexes);
301 }
302
308 constexpr type dtype() const {
309 return source_.dtype();
310 }
311
318 constexpr Shape const& shape() const {
319 return shape_;
320 }
321
328 constexpr Strides const& strides() const {
329 return strides_;
330 }
331
338 std::ptrdiff_t offset() const {
339 return source_.offset();
340 }
341
342 Tensor forward() const;
343
344private:
345 Shape shape_{};
346 Strides strides_{};
347 typename Trait<Source>::Reference source_;
348};
