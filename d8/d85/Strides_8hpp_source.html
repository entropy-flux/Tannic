135 template<Iterator Iterator>
136 constexpr Strides(Iterator begin, Iterator end) {
137 size_type dimension = 0;
138 for (auto iterator = begin; iterator != end; ++iterator) {
139 assert(dimension < limit && "Strides rank limit exceeded");
140 sizes_[dimension++] = static_cast<size_type>(*iterator);
141 }
142 rank_ = dimension;
143 assert(rank_ <= limit && "Strides rank limit exceeded");
144 }

158 constexpr Strides(const Shape& shape) {
159 rank_ = shape.rank();
160 if (rank_ == 0) return;
161
162 sizes_[rank_ - 1] = 1;
163 for (int size = rank_ - 2; size >= 0; --size) {
164 sizes_[size] = sizes_[size + 1] * shape[size + 1];
165 }
166 assert(rank_ <= limit && "Strides rank limit exceeded");
167 }

274constexpr bool operator==(Strides const& first, Strides const& second) {
275 if (first.rank() != second.rank()) return false;
276 for (Strides::rank_type dimension = 0; dimension < first.rank(); ++dimension) {
277 if (first[dimension] != second[dimension]) return false;
278 }
279 return true;
