130 Tensor(type dtype, Shape shape, Strides strides, std::ptrdiff_t offset = 0)
131 : dtype_(dtype)
132 , shape_(shape)
133 , strides_(strides)
134 , offset_(offset) {
135 if (rank() == 0) {
136 nbytes_ = dsizeof(dtype_);
137 }
138 else {
139 std::size_t nbytes = 0;
140 std::size_t expected = 1;
141 for (std::size_t dimension = 0; dimension < rank(); ++dimension) {
142 nbytes += strides_[dimension] * (shape_[dimension] - 1);
143 if (strides_[dimension] != expected) {
144 is_contiguous_ = false;
145 }
146 expected *= shape_[dimension];
147 }
148 nbytes_ = (nbytes + 1) * dsizeof(dtype_);
149 }
150 }
