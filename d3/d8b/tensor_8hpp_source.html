130 Tensor(type dtype, Shape shape, Strides strides, std::ptrdiff_t offset = 0)
131 : dtype_(dtype)
132 , shape_(shape)
133 , strides_(strides)
134 , offset_(offset) {
135 if (rank() == 0) {
136 nbytes_ = dsizeof(dtype_);
137 }
138 else {
139 std::size_t nbytes = 0;
140 std::size_t expected = 1;
141 for (std::size_t dimension = 0; dimension < rank(); ++dimension) {
142 nbytes += strides_[dimension] * (shape_[dimension] - 1);
143 if (strides_[dimension] != expected) {
144 is_contiguous_ = false;
145 }
146 expected *= shape_[dimension];
147 }
148 nbytes_ = (nbytes + 1) * dsizeof(dtype_);
149 }
150 }

451 template<typename T>
452 Tensor& operator=(std::initializer_list<T> const& values) {
453 if (!is_initialized())
454 initialize();
455
456 if (rank() != 1 || shape_[0] != values.size())
457 throw Exception("Shape mismatch in assignment from initializer_list");
458
459 auto fill = [this, &values](auto cast) {
460 using Cast = decltype(cast);
461 size_t index = 0;
462 for (auto value : values) {
463 Cast casted = value;
464 assign(expression::tobytes(casted), index * dsizeof(dtype_));
465 ++index;
466 }
467 };
468
469 switch (dtype_) {
470 case int8: fill(int8_t{}); break;
471 case int16: fill(int16_t{}); break;
472 case int32: fill(int32_t{}); break;
473 case int64: fill(int64_t{}); break;
474 case float32: fill(float{}); break;
475 case float64: fill(double{}); break;
476 default: throw Exception("Unsupported dtype in assignment");
477 }
478
479 return *this;
480 }
