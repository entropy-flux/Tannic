550 template<typename T>
551 void initialize(std::initializer_list<T> values, Environment environment = Host{}) {
552 if (!is_contiguous())
553 throw Exception("Assign to initializer list supported only for contiguous tensors");
554
555 if (rank() != 1 || shape_[0] != values.size())
556 throw Exception("Shape mismatch in assignment from initializer_list");
557
558 if (!is_initialized())
559 initialize(environment);
560
561 if (dtype_ == boolean) {
562 std::ptrdiff_t index = 0;
563 for (auto const& value : values) {
564 assign((bool const*)(&value), index);
565 ++index;
566 }
567 }
568
569 else {
570 auto fill = [this, &values](auto cast) {
571 using Cast = decltype(cast);
572 size_t index = 0;
573 for (auto value : values) {
574 Cast casted = value;
575 assign(expression::tobytes(casted), index * dsizeof(dtype_));
576 ++index;
577 }
578 };
579
580 switch (dtype_) {
581 case int8: fill(int8_t{}); break;
582 case int16: fill(int16_t{}); break;
583 case int32: fill(int32_t{}); break;
584 case int64: fill(int64_t{}); break;
585 case float32: fill(float{}); break;
586 case float64: fill(double{}); break;
587 default: throw Exception("Unsupported dtype in assignment");
588 }
589 }
