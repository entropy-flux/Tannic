131 Tensor(type dtype, Shape shape, Strides strides, std::ptrdiff_t offset = 0)
132 : dtype_(dtype)
133 , shape_(shape)
134 , strides_(strides)
135 , offset_(offset) {
136 if (rank() == 0) {
137 nbytes_ = dsizeof(dtype_);
138 }
139 else {
140 std::size_t nelements = 0;
141 std::size_t expected = 1;
142 for (std::size_t dimension = 0; dimension < rank(); ++dimension) {
143 nelements += strides_[dimension] * (shape_[dimension] - 1);
144 if (strides_[dimension] != expected) {
145 is_contiguous_ = false;
146 }
147 expected *= shape_[dimension];
148 }
149 nbytes_ = nbytesof(dtype, nelements + 1);
150 }

525 template<typename T>
526 Tensor& operator=(std::initializer_list<T> const& values) {
527 if (!is_contiguous())
528 throw Exception("Assign to initializer list supported only for contiguous tensors");
529
530 if (!is_initialized())
531 initialize();
532
533 if (rank() != 1 || shape_[0] != values.size())
534 throw Exception("Shape mismatch in assignment from initializer_list");
535
536 if (dtype_ == boolean) {
537 std::ptrdiff_t index = 0;
538 for (auto const& value : values) {
539 assign((bool const*)(&value), index);
540 ++index;
541 }
542 }
543
544 else {
545 auto fill = [this, &values](auto cast) {
546 using Cast = decltype(cast);
547 size_t index = 0;
548 for (auto value : values) {
549 Cast casted = value;
550 assign(expression::tobytes(casted), index * dsizeof(dtype_));
551 ++index;
552 }
553 };
554
555 switch (dtype_) {
556 case int8: fill(int8_t{}); break;
557 case int16: fill(int16_t{}); break;
558 case int32: fill(int32_t{}); break;
559 case int64: fill(int64_t{}); break;
560 case float32: fill(float{}); break;
561 case float64: fill(double{}); break;
562 default: throw Exception("Unsupported dtype in assignment");
563 }
564 }
565
566 return *this;
567
