84template <Expression Source, class... Indexes>
85class Slice {
86public:
87
96 constexpr Slice(typename Trait<Source>::Reference source, std::tuple<Indexes...> indexes)
97 : dtype_(source.dtype())
98 , source_(source)
99 , indexes_(indexes)
100 {
101 std::array<Shape::size_type, Shape::limit> shape{};
102 std::array<Strides::size_type, Strides::limit> strides{};
103 Shape::rank_type dimension = 0;
104 Shape::rank_type rank = 0;
105 offset_ = 0;
106 auto process = [&](const auto& argument) {
107 using Argument = std::decay_t<decltype(argument)>;
108 if constexpr (std::is_same_v<Argument, indexing::Range>) {
109 auto range = normalize(argument, source.shape()[dimension]);
110 auto size = range.stop - range.start;
111 assert(size >= 0);
112 shape[rank] = size;
113 strides[rank] = source.strides()[dimension];
114 offset_ += range.start * source.strides()[dimension] * dsizeof(dtype_);
115 rank++; dimension++;
116 }
117
118 else if constexpr (std::is_integral_v<Argument>) {
119 auto index = indexing::normalize(argument, source.shape()[dimension]);
120 offset_ += index * source.strides()[dimension] * dsizeof(dtype_);
121 dimension++;
122 }
123
124 else {
125 static_assert(sizeof(Argument) == 0, "Unsupported index type in Slice");
126 }
127 };
128
129 std::apply([&](const auto&... arguments) {
130 (process(arguments), ...);
131 }, indexes);
132
133 while (dimension < source.rank()) {
134 shape[rank] = source.shape()[dimension];
135 strides[rank] = source.strides()[dimension];
136 rank++; dimension++;
137 }
138
139 shape_ = Shape(shape.begin(), shape.begin() + rank);
140 strides_ = Strides(strides.begin(), strides.begin() + rank);
141 }

302template <Expression Source, class... Indexes>
303template <typename T>
304void Slice<Source, Indexes...>::operator=(T value) {
305 auto copy = [this](std::byte const* value, std::ptrdiff_t offset) {
306 if(rank() == 0) {
307 assign(value, offset);
308 return;
309 }
310 std::vector<std::size_t> indexes(rank(), 0);
311 bool done = false;
312
313 while (!done) {
314 std::size_t position = offset;
315 for (auto dimension = 0; dimension < rank(); ++dimension) {
316 position += indexes[dimension] * strides_[dimension] * dsizeof(dtype_);
317 }
318
319 assign(value, position);
320 done = true;
321 for (int dimension = rank() - 1; dimension >= 0; --dimension) {
322 if (++indexes[dimension] < shape_[dimension]) {
323 done = false;
324 break;
325 }
326 indexes[dimension] = 0;
327 }
328 }
329 };
330
331 switch (dtype_) {
332 case int8: {
333 int8_t casted = value;
334 copy(tobytes(casted), offset());
335 break;
336 }
337 case int16: {
338 int16_t casted = value;
339 copy(tobytes(casted), offset());
340 break;
341 }
342 case int32: {
343 int32_t casted = value;
344 copy(tobytes(casted), offset());
345 break;
346 }
347 case int64: {
348 int64_t casted = value;
349 copy(tobytes(casted), offset());
350 break;
351 }
352 case float32: {
353 float casted = value;
354 copy(tobytes(casted), offset());
355 break;
356 }
357 case float64: {
358 double casted = value;
359 copy(tobytes(casted), offset());
360 break;
361 }
362 default:
363 break;
364 }
365}
