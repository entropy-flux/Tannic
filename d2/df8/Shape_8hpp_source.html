284constexpr bool operator==(Shape const& first, Shape const& second) {
285 if (first.rank() != second.rank()) return false;
286 for (Shape::size_type dimension = 0; dimension < first.rank(); ++dimension) {
287 if (first[dimension] != second[dimension]) return false;
288 }
289 return true;
