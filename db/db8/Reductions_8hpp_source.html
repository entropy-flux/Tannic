75template<class Reducer, Expression Operand>
76class Reduction {
77public:
78 Reducer reducer;
79 typename Trait<Operand>::Reference operand;
80
81 constexpr Reduction(Reducer reducer, typename Trait<Operand>::Reference operand)
82 : reducer(reducer)
83 , operand(operand)
84 , dtype_(reducer.reduce(operand.dtype()))
85 , shape_(reducer.reduce(operand.shape()))
86 , strides_(shape_) {}
87
88 constexpr type dtype() const {
89 return dtype_;
90 }
91
92 constexpr Shape const& shape() const {
93 return shape_;
94 }
95
96 constexpr Strides const& strides() const {
97 return strides_;
98 }
99
100 std::ptrdiff_t offset() const {
101 return 0;
102 }
103
104 Tensor forward() const {
105 Tensor source = operand.forward();
106 Tensor result(dtype(), shape(), strides(), offset());
107 reducer.forward(source, result);
108 return result;
109 }
110
111private:
112 type dtype_;
113 Shape shape_;
114 Strides strides_;
115};

138struct Argmax {
139 int axis;
140 bool keepdim;
141
142 constexpr type reduce(type dtype) const {
143 return int64;
144 }
145
146 constexpr Shape reduce(Shape const& shape) const {
147 if (shape.rank() == 0)
148 throw Exception("Cannot reduce scalar tensors");
149
150 Shape out;
151 for (size_t dim = 0; dim < shape.rank(); ++dim) {
152 if (dim != static_cast<size_t>(axis)) out.expand(shape[dim]);
153 else if (keepdim) out.expand(1);
154 }
155 return out;
156 }
157
158 void forward(Tensor const& input, Tensor& output) const;
159};
