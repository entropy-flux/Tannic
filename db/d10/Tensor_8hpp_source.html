129 Tensor(type dtype, Shape shape, Strides strides, std::ptrdiff_t offset = 0)
130 : dtype_(dtype)
131 , shape_(shape)
132 , strides_(strides)
133 , offset_(offset) {
134 if (rank() == 0) {
135 nbytes_ = dsizeof(dtype_);
136 }
137 else {
138 std::size_t nbytes = 0;
139 std::size_t expected = 0;
140 for (std::size_t dimension = 0; dimension < rank(); ++dimension) {
141 nbytes += strides_[dimension] * (shape_[dimension] - 1);
142 }
143 nbytes_ = (nbytes + 1) * dsizeof(dtype_);
144 }
145 }
