45template<class Coordinates, Expression Source>
46class Complexification<Coordinates, Source> {
47public:
48 typename Trait<Source>::Reference source;
49
50 constexpr Complexification(Trait<Source>::Reference source)
51 : source(source)
52 {
53 switch (source.dtype()) {
54 case float32: dtype_ = complex64; break;
55 case float64: dtype_ = complex128; break;
56 default: assert(false && "Complex view error: source tensor dtype must be float32 or float64");
57 }
58
59 if (source.strides()[-1] == 1 && source.strides()[-2] == 2) {
60 assert(source.shape().back() == 2 &&
61 "Complex view error: last dimension must be size 2 (real + imag).");
62 shape_ = Shape(source.shape().begin(), source.shape().end() - 1);
63 strides_ = Strides(source.strides().begin(), source.strides().end() - 1);
64 strides_[-1] = 1;
65 } else {
66 assert(false &&
67 "Complex view error: source tensor is not contiguous in last two dimensions. "
68 "Cannot create complex view safely."
69 );
70 }
71 }
72
73 constexpr type dtype() const {
74 return dtype_;
75 }
76
77 constexpr Shape const& shape() const {
78 return shape_;
79 }
80
81 constexpr Strides const& strides() const {
82 return strides_;
83 }
84
85 std::ptrdiff_t offset() const {
86 return source.offset();
87 }
88
89 Tensor forward() const;
90
91private:
92 type dtype_;
93 Shape shape_;
94 Strides strides_;
95};

97template<class Coordinates, Expression Real, Expression Imaginary>
98class Complexification<Coordinates, Real, Imaginary> {
99public:
100 typename Trait<Real>::Reference real;
101 typename Trait<Imaginary>::Reference imaginary;
102
103 constexpr Complexification(typename Trait<Real>::Reference real, typename Trait<Imaginary>::Reference imaginary)
104 : real(real)
105 , imaginary(imaginary)
106 {
107 assert(real.shape() == imaginary.shape() && "Complexification error: real and imaginary part shapes must match");
108 assert(real.strides() == imaginary.strides() && "Complexification error: real and imaginary part shapes must match");
109 if (real.dtype() == float64 || imaginary.dtype() == float64) {
110 dtype_ = complex128;
111 } else {
112 dtype_ = complex64;
113 }
114 }

139template<Expression Source>
140class Realification {
141public:
142 typename Trait<Source>::Reference source;
143
144 constexpr Realification(Trait<Source>::Reference source)
145 : source(source)
146 {
147 switch (source.dtype()) {
148 case complex64: dtype_ = float32; break;
149 case complex128: dtype_ = float64; break;
150 default:
151 assert(false &&
152 "Real view error: source tensor dtype must be complex64 or complex128");
153 }
154
155 if (source.strides()[-1] == 1) {
156 shape_ = Shape(source.shape().begin(), source.shape().end());
157 shape_.expand(2);
158
159 strides_ = Strides(source.strides().begin(), source.strides().end());
160 strides_.expand(1);
161 strides_[-2] = 2;
162 } else {
163 assert(false && "Real view error: source tensor is not in interleaved real/imag format");
164 }
165
166 }
167
168 constexpr type dtype() const {
169 return dtype_;
170 }
171
172 constexpr Shape const& shape() const {
173 return shape_;
174 }
175
176 constexpr Strides const& strides() const {
177 return strides_;
178 }
179
180 std::ptrdiff_t offset() const {
181 return source.offset();
182 }
183
184 Tensor forward() const;
185
186private:
187 type dtype_;
188 Shape shape_;
189 Strides strides_;
190};

192template<Expression Real>
193constexpr auto complex(Real&& real) {
194 return Complexification<Cartesian, Real>{std::forward<Real>(real)};
195}

197template<Expression Real, Expression Imaginary>
198constexpr auto complex(Real&& real, Imaginary&& imaginary) {
199 return Complexification<Cartesian, Real, Imaginary>{std::forward<Real>(real), std::forward<Imaginary>(imaginary)};
200}
