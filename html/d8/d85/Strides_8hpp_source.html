135 template<Iterator Iterator>
136 constexpr Strides(Iterator begin, Iterator end) {
137 size_type dimension = 0;
138 for (auto iterator = begin; iterator != end; ++iterator) {
139 assert(dimension < limit && "Strides rank limit exceeded");
140 sizes_[dimension++] = static_cast<size_type>(*iterator);
141 }
142 rank_ = dimension;
143 }

157 constexpr Strides(const Shape& shape) {
158 rank_ = shape.rank();
159 if (rank_ == 0) return;
160
161 sizes_[rank_ - 1] = 1;
162 for (int size = rank_ - 2; size >= 0; --size) {
163 sizes_[size] = sizes_[size + 1] * shape[size + 1];
164 }
165 }

266constexpr bool operator==(Strides const& first, Strides const& second) {
267 if (first.rank() != second.rank()) return false;
268 for (Strides::rank_type dimension = 0; dimension < first.rank(); ++dimension) {
269 if (first[dimension] != second[dimension]) return false;
270 }
271 return true;
272}
