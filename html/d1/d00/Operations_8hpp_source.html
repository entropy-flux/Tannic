265static constexpr Shape broadcast(Shape const& first, Shape const& second) {
266 auto first_rank = first.rank();
267 auto second_rank = second.rank();
268 auto rank = std::max(first_rank, second_rank);
269 std::vector<std::size_t> result(rank, 1);
270
271 for (auto dimension = 0; dimension < rank; ++dimension) {
272 auto first_dimension = (dimension < rank - first_rank) ? 1 : first[dimension - (rank - first_rank)];
273 auto second_dimension = (dimension < rank - second_rank) ? 1 : second[dimension - (rank - second_rank)];
274
275 if (!(first_dimension == second_dimension || first_dimension == 1 || second_dimension == 1)) {
276 throw std::invalid_argument("Shapes are not broadcast-compatible.");
277 }
278 result[dimension] = std::max(first_dimension, second_dimension);
279 }
280 return Shape(result);
281}
