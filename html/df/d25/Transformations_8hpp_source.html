241 static constexpr Shape broadcast(Shape const& first, Shape const& second) {
242 auto first_rank = first.rank();
243 auto second_rank = second.rank();
244
245 // Vector dot product
246 if (first_rank == 1 && second_rank == 1) {
247 assert(first[0] == second[0] && "Vector dimensions must match for dot product");
248 return Shape{}; // Scalar result
249 }
250
251 // Matrix-vector multiplication
252 if (first_rank == 1 && second_rank == 2) {
253 assert(first[0] == second[1] && "Matrix inner dimensions do not match");
254 return Shape{second[0]}; // Vector result
255 }
256
257 // Vector-matrix multiplication
258 if (first_rank == 2 && second_rank == 1) {
259 assert(first[1] == second[0] && "Matrix inner dimensions do not match");
260 return Shape{first[0]}; // Vector result
261 }
262
263 assert(first_rank >= 2 && second_rank >= 2 && "Inputs must have rank >= 2");
264
265 // Handle batch dimensions
266 Shape first_batches(first.begin(), first.end() - 2);
267 Shape second_batches(second.begin(), second.end() - 2);
268 Shape batches = operation::broadcast(first_batches, second_batches);
269
270 // Check matrix inner dimensions
271 auto K1 = *(first.end() - 1);
272 auto K2 = *(second.end() - 2);
273 assert(K1 == K2 && "Inner dimensions must match for matmul");
274
275 // Get output matrix dimensions
276 auto M = *(first.end() - 2);
277 auto N = *(second.end() - 1);
278
279 // Combine batch and matrix dimensions
280 std::vector<Shape::size_type> result(batches.begin(), batches.end());
281 result.push_back(M);
282 result.push_back(N);
283 return Shape(result);
284 }
285};
