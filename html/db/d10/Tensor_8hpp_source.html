127 Tensor(type dtype, Shape shape, Strides strides, std::ptrdiff_t offset = 0)
128 : dtype_(dtype)
129 , shape_(shape)
130 , strides_(strides)
131 , offset_(offset) {
132 if (rank() == 0) {
133 nbytes_ = dsizeof(dtype_);
134 }
135 else {
136 std::size_t nbytes = 0;
137 std::size_t expected = 0;
138 for (std::size_t dimension = 0; dimension < rank(); ++dimension) {
139 nbytes += strides_[dimension] * (shape_[dimension] - 1);
140 }
141 nbytes_ = (nbytes + 1) * dsizeof(dtype_);
142 }
143 }
