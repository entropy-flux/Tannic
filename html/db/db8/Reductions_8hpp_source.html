83 constexpr Reduction(Reducer reducer, typename Trait<Operand>::Reference operand)

126 Tensor forward() const {

145struct Argmax {
146 int axis;
147
153 constexpr type reduce(type dtype) const {
154 return int64;
155 }
156
163 constexpr Shape reduce(Shape const& shape) const {
164 assert(shape.rank() >= 1);
165 std::vector<size_t> sizes;
166 sizes.reserve(shape.rank() - 1);
167 for (size_t dimension = 0; dimension < shape.rank(); ++dimension) {
168 if (dimension != static_cast<size_t>(axis))
169 sizes.push_back(shape[dimension]);
170 }
171 return Shape(sizes.begin(), sizes.end());
172 }
173
179 constexpr Strides reduce(Strides const& strides) const {
180 if (strides.rank() == 1) {
181 return Strides();
182 }
183 std::vector<Strides::size_type> sizes;
184 sizes.reserve(strides.rank() - 1);
185 for (size_t dimension = 0; dimension < strides.rank(); ++dimension) {
186 if (dimension != static_cast<size_t>(axis))
187 sizes.push_back(strides[dimension]);
188 }
189 return Strides(sizes.begin(), sizes.end());
190 }
191
197 void forward(Tensor const&, Tensor&) const;
198};

287template<Expression Source>
288constexpr auto argmax(Source&& source, int axis = -1) {
289 return Reduction<Argmax, Source>{
290 {indexing::normalize(axis, source.shape().rank())}, std::forward<Source>(source)
291 };
292}
