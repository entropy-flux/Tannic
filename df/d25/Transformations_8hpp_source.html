214 constexpr static type promote(type inner, type outer) {
215 type dtype = promotions[index(inner, outer)];
216 if (dtype == none)
217 throw Exception("Unsuported dtypes");

235 static constexpr Shape transform(Shape const& first, Shape const& second) {
236 auto first_rank = first.rank();
237 auto second_rank = second.rank();
238
239 if (first_rank == 1 && second_rank == 1) {
240 if (first_rank != 1 | second_rank != 1)
241 throw Exception("dimensions must match for dot product");
242 return Shape{}; // Scalar result
243 }
244
245 if (first_rank == 1 && second_rank == 2) {
246 if (first[0] != second[0])
247 throw Exception("Matrix inner dimensions do not match");
248 if (first[0] != second[0])
249 return Shape{second[0]}; // Vector result
250 }
251
252 // Vector-matrix multiplication
253 if (first_rank == 2 && second_rank == 1) {
254 if (first[1] != second[0])
255 throw Exception("Matrix inner dimensions do not match");
256 return Shape{first[0]}; // Vector result
257 }
258
259 if (first_rank < 2 | second_rank < 2)
260 throw Exception("Inputs must have rank >= 2");
261
262 // Handle batch dimensions
263 Shape first_batches(first.begin(), first.end() - 2);
264 Shape second_batches(second.begin(), second.end() - 2);
265 Shape batches = operation::broadcast(first_batches, second_batches);
266
267 // Check matrix inner dimensions
268 auto K1 = *(first.end() - 1);
269 auto K2 = *(second.end() - 2);
270 assert(K1 == K2 && "Inner dimensions must match for matmul");
271
272 // Get output matrix dimensions
273 auto M = *(first.end() - 2);
274 auto N = *(second.end() - 1);
275
276 // Combine batch and matrix dimensions
277 std::vector<Shape::size_type> result(batches.begin(), batches.end());
278 result.push_back(M);
279 result.push_back(N);
280 return Shape(result);
281 }

283 void forward(Tensor const& outer, Tensor const& inner, Tensor& result) const;
284};

372struct Concatenation {
373 int axis;
383 constexpr auto promote(type first, type second) const {
384 if(first != second)
385 throw Exception("Cannot concatenate tensors of different dtypes");
386 return first;
387 }
388
399 constexpr Shape transform(Shape const& first, Shape const& second) {
400 assert(first.rank() == second.rank() && "Ranks must match for concatenation");
401 Shape result = first;
402
403 for (int dimension = 0; dimension < first.rank(); ++dimension) {
404 if (dimension == axis) {
405 result[dimension] = first[dimension] + second[dimension];
406 } else {
407 if (first[dimension] != second[dimension])
408 throw Exception("All dimensions except concat axis must match");
409 }
410 }
411
412 return result;
413 }
414
415 void forward(Tensor const&, Tensor const&, Tensor&) const;
416};
