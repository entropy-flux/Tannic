233 static constexpr Shape transform(Shape const& first, Shape const& second) {
234 auto first_rank = first.rank();
235 auto second_rank = second.rank();
236
237 // Vector dot product
238 if (first_rank == 1 && second_rank == 1) {
239 assert(first[0] == second[0] && "Vector dimensions must match for dot product");
240 return Shape{}; // Scalar result
241 }
242
243 // Matrix-vector multiplication
244 if (first_rank == 1 && second_rank == 2) {
245 assert(first[0] == second[1] && "Matrix inner dimensions do not match");
246 return Shape{second[0]}; // Vector result
247 }
248
249 // Vector-matrix multiplication
250 if (first_rank == 2 && second_rank == 1) {
251 assert(first[1] == second[0] && "Matrix inner dimensions do not match");
252 return Shape{first[0]}; // Vector result
253 }
254
255 assert(first_rank >= 2 && second_rank >= 2 && "Inputs must have rank >= 2");
256
257 // Handle batch dimensions
258 Shape first_batches(first.begin(), first.end() - 2);
259 Shape second_batches(second.begin(), second.end() - 2);
260 Shape batches = operation::broadcast(first_batches, second_batches);
261
262 // Check matrix inner dimensions
263 auto K1 = *(first.end() - 1);
264 auto K2 = *(second.end() - 2);
265 assert(K1 == K2 && "Inner dimensions must match for matmul");
266
267 // Get output matrix dimensions
268 auto M = *(first.end() - 2);
269 auto N = *(second.end() - 1);
270
271 // Combine batch and matrix dimensions
272 std::vector<Shape::size_type> result(batches.begin(), batches.end());
273 result.push_back(M);
274 result.push_back(N);
275 return Shape(result);
276 }

278 void forward(Tensor const& outer, Tensor const& inner, Tensor& result) const;
